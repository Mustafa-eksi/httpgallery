33std::string read_binary_to_string(const std::string path,
34 uintmax_t range_start = 0,
35 uintmax_t range_end = 0)
36{
37 if (range_end < range_start)
38 return "error: range";
39 if (range_end == 0)
40 range_end = std::filesystem::file_size(path);
41
42 std::ifstream file(path, std::ios::binary);
43 if (!file) {
44 std::cout << "ERROR!: " << path << std::endl;
45 return "Error";
46 }
47 file.seekg(range_start);
48 std::string strbuff(range_end - range_start, '\0');
49 if (!file.read(&strbuff[0], range_end - range_start)) {
50 strbuff.clear();
51 strbuff.shrink_to_fit();
52 return "Error";
53 }
54 return strbuff;
55}

79 std::string read(const std::string path, uintmax_t range_start = 0,
80 uintmax_t range_end = 0)
81 {
82 auto cache_entry_name = path;
83 auto opt_val = cache.get(cache_entry_name);
84 if (opt_val) {
85 auto file_entry = opt_val.value();
86 auto r0 = file_entry.range.first;
87 auto r1 = file_entry.range.second;
88 logger.changeMetric("Partial Cache Hit", 1);
89 // r0 p0 p1 r1
90 // If Cached value includes the part we want:
91 if (r0 <= range_start && range_end <= r1) {
92 return file_entry.data.substr(
93 range_start - r0, (range_end - r0) - (range_start - r0));
94 }
95 // r0 p0 r1 p1
96 if (r0 <= range_start && range_end < r1) {
97 std::string strbuff
98 = read_binary_to_string(path, r1, range_end);
99 std::string new_data
100 = file_entry.data.substr(range_start - r0) + strbuff;
101 cache.put(cache_entry_name, new_data,
102 std::make_pair(r0, range_end));
103 return new_data;
104 }
105 // p0 r0 p1 r1
106 if (range_start <= r0 && range_end <= r1) {
107 std::string strbuff
108 = read_binary_to_string(path, range_start, r0);
109 std::string new_data
110 = strbuff + file_entry.data.substr(0, range_end - r0);
111 cache.put(cache_entry_name, new_data,
112 std::make_pair(range_start, r1));
113 return new_data;
114 }
115 // p0 r0 r1 p1
116 if (range_start <= r0 && r1 <= range_end) {
117 std::string left_part
118 = read_binary_to_string(path, range_start, r0);
119 std::string right_part
120 = read_binary_to_string(path, r1, range_end);
121 std::string new_data = left_part + file_entry.data + right_part;
122 cache.put(cache_entry_name, new_data,
123 std::make_pair(range_start, range_end));
124 return new_data;
125 }
126 // TODO: Cache non-intersecting ranges
127 }
128 logger.changeMetric("Cache Miss", 1);
129 std::string strbuff
130 = read_binary_to_string(path, range_start, range_end);
131 cache.put(cache_entry_name, strbuff,
132 std::make_pair(range_start, range_end));
133 return strbuff;
134 }

154 std::string queries = "", bool list_view = false)

156 std::string output = "";
157 for (const auto &entry : std::filesystem::directory_iterator(path)) {
158 try {
159 if (!entry.is_regular_file() && !entry.is_directory())
160 continue;
161 if (current_address.back() != '/')
162 current_address += '/';
163
164 auto filename = entry.path().filename().string();
165 auto filepath = current_address + filename + queries;
166 auto iconpath = filepath;
167
168 if (!queries.empty()) {
169 iconpath += "&icon=1";
170 } else {
171 iconpath += "?icon=1";
172 }
173
174 if (get_mime_type(filename).starts_with("image") && !list_view) {
175 output += string_format(imageTemplate, filepath, filepath,
176 filename);
177 } else if (get_mime_type(filename).starts_with("video")
178 && entry.file_size() < 5e+7 && !list_view) {
179 output += string_format(videoTemplate, video_icon_link,
180 filepath, filename);
181 } else {
182 output += string_format(buttonTemplate,
183 entry.is_directory() ? dir_icon_link
184 : text_icon_link,
185 filepath, filename);
186 }
187 } catch (std::exception &e) {
188 std::cout << "Error: " << entry.path().c_str() << e.what()
189 << std::endl;
190 }
191 }

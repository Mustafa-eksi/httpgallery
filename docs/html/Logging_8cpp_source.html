46void Logger::exportMetrics()
47{
48 auto datfile = std::ofstream("metrics.dat");
49 auto logs = std::ifstream(output_filename);
50 std::string line;
51 int start_timestamp = 0;
52 std::unordered_map<std::string, std::string> datas;
53 while (getline(logs, line)) {
54 if (line.ends_with("Starting Server")) {
55 auto timestamp_del = line.find("@ ");
56 auto metadata_end_del = line.find("]");
57 if (timestamp_del == std::string::npos)
58 continue;
59 auto time_str = line.substr(timestamp_del + 2,
60 metadata_end_del - timestamp_del - 2);
61 start_timestamp = std::stoi(time_str);
62 } else if (line.starts_with("[METRICS")) {
63 auto timestamp_del = line.find("@ ");
64 auto keyval_del = line.find("]: ");
65 if (keyval_del == std::string::npos
66 || timestamp_del == std::string::npos)
67 continue;
68 auto timestamp = line.substr(timestamp_del + 2,
69 keyval_del - timestamp_del - 2);
70 auto keyval = line.substr(keyval_del + 3);
71 int time_int = 0;
72 try {
73 time_int = std::stoi(timestamp);
74 } catch (std::invalid_argument &e) {
75 std::cout << "Error: " << timestamp << ", " << e.what()
76 << std::endl;
77 return;
78 }
79 auto key_del = keyval.find("=");
80 if (key_del == std::string::npos)
81 return;
82 auto key = keyval.substr(0, key_del);
83 auto val = keyval.substr(key_del + 1);
84 if (!datas.contains(key))
85 datas[key] = std::to_string(time_int - start_timestamp) + " "
86 + val + "\n";
87 else
88 datas[key] += std::to_string(time_int - start_timestamp) + " "
89 + val + "\n";
90 }
91 }
92 size_t i = 0;
93 for (auto &[key, valpair] : datas) {
94 datfile << "\"" << key << "\"" << std::endl;
95 if (i++ == datas.size() - 1)
96 datfile << valpair;
97 else
98 datfile << valpair << std::endl << std::endl;
99 }
100 datfile.close();
101}

44HttpResponseBuilder HttpResponseBuilder::CompressContent(std::string encoding)
45{
46 // TODO: support brotli
47 if (encoding.find("gzip") == std::string::npos
48 && encoding.find("*") == std::string::npos) {
49 return *this;
50 }
51 if (this->content.empty())
52 return *this;
53 auto destLen = compressBound(this->content.length());
54 std::string destBuffer(destLen, '\0');
55 z_stream_s zs = {};
56 zs.next_in = (unsigned char *)this->content.data();
57 zs.avail_in = this->content.length();
58 zs.next_out = (unsigned char *)destBuffer.data();
59 zs.avail_out = destLen;
60 deflateInit2(&zs, Z_DEFAULT_COMPRESSION, Z_DEFLATED, 15 + 16, 8,
61 Z_DEFAULT_STRATEGY);
62 deflate(&zs, Z_FINISH);
63 deflateEnd(&zs);
64
65 // destBuffer might be leaking
66 return this->Content(destBuffer).SetHeader("Content-Encoding", "gzip");
67}
